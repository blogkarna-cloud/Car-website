    make VARCHAR(100),
    model VARCHAR(100),
    variant VARCHAR(200),
    year INTEGER,
    body_type VARCHAR(50),
    fuel_type VARCHAR(30),
    transmission VARCHAR(30),
    price DECIMAL(10,2),
